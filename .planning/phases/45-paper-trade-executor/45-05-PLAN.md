**run_paper_executor.py:**
CLI entry point following project pattern (argparse + engine creation + PaperExecutor.run).

```python
#!/usr/bin/env python
"""
Paper-trade executor CLI.

Reads signals for active strategies, generates paper orders, simulates fills,
and updates positions. Runs for all active configs in dim_executor_config.

Usage:
    # Standard run (process new signals for all active strategies)
    python -m ta_lab2.scripts.executor.run_paper_executor

    # Dry run (log decisions without writing to DB)
    python -m ta_lab2.scripts.executor.run_paper_executor --dry-run

    # Replay historical signals for backtest parity check
    python -m ta_lab2.scripts.executor.run_paper_executor --replay-historical --start 2024-01-01 --end 2025-01-01

    # Verbose output
    python -m ta_lab2.scripts.executor.run_paper_executor --verbose
"""
```

Arguments:
- --db-url: Database URL (default: from config/env via resolve_db_url)
- --dry-run: Log decisions without writing
- --verbose: Detailed output
- --replay-historical: Replay mode for backtest parity
- --start: Start date for replay (required with --replay-historical)
- --end: End date for replay (required with --replay-historical)

Main logic:
1. Parse args
2. Resolve DB URL via resolve_db_url
3. Create engine with NullPool (project convention)
4. Instantiate PaperExecutor(engine)
5. Call executor.run(dry_run=args.dry_run, replay_historical=args.replay_historical, ...)
6. Print summary
7. Exit 0 on success, 1 on failure

```python
def run_signal_refreshes(args, db_url: str) -> ComponentResult:
    """Run signal generation via subprocess."""
    cmd = [
        sys.executable, "-m",
        "ta_lab2.scripts.signals.run_all_signal_refreshes",
    ]
    if getattr(args, "dry_run", False):
        # Signal refresher doesn't have --dry-run; skip entirely
        print(f"\n{'=' * 70}")
        print("RUNNING SIGNAL GENERATION")
        print(f"{'=' * 70}")
        print("[DRY RUN] Would run signal generation")
        return ComponentResult("signals", True, 0.0, 0)

    print(f"\n{'=' * 70}")
    print("RUNNING SIGNAL GENERATION")
    print(f"{'=' * 70}")
    print(f"Command: {' '.join(cmd)}")

    start = time.perf_counter()
    try:
        result = subprocess.run(
            cmd, check=False, capture_output=True, text=True, timeout=TIMEOUT_SIGNALS
        )
        duration = time.perf_counter() - start

        if result.returncode != 0:
            print(f"\n[ERROR] Signal generation failed (code {result.returncode})")
            if result.stdout:
                print(f"\nSTDOUT:\n{result.stdout[-2000:]}")
            if result.stderr:
                print(f"\nSTDERR:\n{result.stderr[-2000:]}")
            return ComponentResult("signals", False, duration, result.returncode,
                                   f"Exited with code {result.returncode}")

        print(f"\n[OK] Signal generation completed in {duration:.1f}s")
        return ComponentResult("signals", True, duration, result.returncode)

    except subprocess.TimeoutExpired:
        duration = time.perf_counter() - start
        return ComponentResult("signals", False, duration, -1,
                               f"Timed out after {TIMEOUT_SIGNALS}s")
```
