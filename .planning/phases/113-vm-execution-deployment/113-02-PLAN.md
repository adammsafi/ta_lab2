Create `PriceCache` class as described in RESEARCH.md Pattern 1:

```python
class PriceCache:
    def __init__(self):
        self._prices: dict[str, Decimal] = {}
        self._timestamps: dict[str, datetime] = {}
        self._lock = threading.RLock()
    
    def update(self, symbol: str, price: float) -> None:
        """Update price for symbol. Thread-safe."""
        with self._lock:
            self._prices[symbol] = Decimal(str(price))
            self._timestamps[symbol] = datetime.now(timezone.utc)
    
    def get(self, symbol: str) -> Decimal | None:
        """Get latest price for symbol. Returns None if not available."""
        with self._lock:
            return self._prices.get(symbol)
    
    def get_with_age(self, symbol: str) -> tuple[Decimal | None, float]:
        """Get price and age in seconds. Returns (None, inf) if unavailable."""
        with self._lock:
            price = self._prices.get(symbol)
            ts = self._timestamps.get(symbol)
            if price is None or ts is None:
                return None, float('inf')
            age = (datetime.now(timezone.utc) - ts).total_seconds()
            return price, age
    
    def is_stale(self, symbol: str, max_age_seconds: float = 120.0) -> bool:
        """Check if price is older than max_age_seconds."""
        _, age = self.get_with_age(symbol)
        return age > max_age_seconds
    
    def all_symbols(self) -> list[str]:
        """Return list of all symbols with prices."""
        with self._lock:
            return list(self._prices.keys())
    
    def stale_symbols(self, max_age_seconds: float = 120.0) -> list[str]:
        """Return symbols with stale or missing prices."""
        with self._lock:
            now = datetime.now(timezone.utc)
            stale = []
            for sym, ts in self._timestamps.items():
                if (now - ts).total_seconds() > max_age_seconds:
                    stale.append(sym)
            return stale
```

All arithmetic uses Decimal (financial precision). Timestamps track when each price was last updated for staleness detection.
