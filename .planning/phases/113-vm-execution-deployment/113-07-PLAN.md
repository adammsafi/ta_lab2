**1. Create ta-executor.service (Pattern 8 from RESEARCH.md):**

```ini
[Unit]
Description=TA Lab2 Paper Executor
After=network-online.target postgresql.service
StartLimitIntervalSec=300
StartLimitBurst=5

[Service]
Type=simple
User=ubuntu
WorkingDirectory=/home/ubuntu/executor
EnvironmentFile=/home/ubuntu/executor/.env
ExecStart=/home/ubuntu/executor/venv/bin/python executor_service.py
Restart=on-failure
RestartSec=30
StandardOutput=journal
StandardError=journal
SyslogIdentifier=ta-executor
TimeoutStopSec=30

[Install]
WantedBy=multi-user.target
```

**CRITICAL:** StartLimitBurst and StartLimitIntervalSec MUST be in [Unit] section, NOT [Service]. Silently ignored in [Service] (from RESEARCH.md).

**2. Create setup_vm.sh (VM-side one-time setup):**

Follow the pattern of deploy/tvc/setup_vm.sh:

```bash
#!/usr/bin/env bash
set -euo pipefail
echo "=== TA Lab2 Executor VM Setup ==="

# 1. Create directory
mkdir -p /home/ubuntu/executor
cd /home/ubuntu/executor

# 2. Python venv
if [ ! -d venv ]; then
    python3 -m venv venv
fi
source venv/bin/activate
pip install --upgrade pip

# 3. Install requirements
pip install -r requirements.txt

# 4. Install ta_lab2 package
# The ta_lab2 source tree must be available. Options:
# a) SCP the whole src/ tree and pip install -e .
# b) Build wheel locally, SCP, pip install *.whl
# For now, use editable install from SCP'd source:
pip install -e /home/ubuntu/executor/ta_lab2_src/

# 5. Create .env if not exists
if [ ! -f .env ]; then
    cat > .env <<'ENVEOF'
EXECUTOR_DB_URL=postgresql+psycopg2://hluser:hlpass@127.0.0.1:5432/hyperliquid
TELEGRAM_BOT_TOKEN=8590137517:AAFy7aSm05SInNOsUx6kmD6eWaLCgtkR2kg
TELEGRAM_CHAT_ID=5591688420
KRAKEN_SYMBOLS=
COINBASE_PRODUCTS=
LOG_LEVEL=INFO
ENVEOF
    echo "Created .env"
fi

# 6. Install systemd service
sudo cp ta-executor.service /etc/systemd/system/
sudo systemctl daemon-reload
sudo systemctl enable ta-executor
echo "Systemd service installed and enabled"

echo "=== Setup complete ==="
echo "Start with: sudo systemctl start ta-executor"
echo "Check status: sudo systemctl status ta-executor"
echo "View logs: journalctl -u ta-executor -f"
```

**3. Create deploy.sh (local-side one-command deploy):**

```bash
#!/usr/bin/env bash
set -euo pipefail
echo "=== Deploying Executor to Oracle VM ==="

SSH_KEY=~/Downloads/oracle_sg_keys/ssh-key-2026-03-10.key
VM_USER=ubuntu
VM_HOST=161.118.209.59
VM_DIR=/home/ubuntu/executor

# 1. SCP deploy files
echo "Copying deploy files..."
scp -i $SSH_KEY -o StrictHostKeyChecking=accept-new \
    deploy/executor/executor_service.py \
    deploy/executor/requirements.txt \
    deploy/executor/setup_vm.sh \
    deploy/executor/ta-executor.service \
    deploy/executor/vm_table_list.txt \
    deploy/executor/create_vm_tables.sh \
    $VM_USER@$VM_HOST:$VM_DIR/

# 2. SCP ta_lab2 source (for editable install)
echo "Copying ta_lab2 source..."
scp -i $SSH_KEY -r \
    src/ta_lab2/ \
    pyproject.toml \
    $VM_USER@$VM_HOST:$VM_DIR/ta_lab2_src/

# 3. Run setup on VM
echo "Running VM setup..."
ssh -i $SSH_KEY $VM_USER@$VM_HOST "cd $VM_DIR && chmod +x setup_vm.sh && ./setup_vm.sh"

# 4. Create tables (run separately -- needs local DB access)
echo ""
echo "=== NEXT STEPS ==="
echo "1. Run table creation:  bash deploy/executor/create_vm_tables.sh"
echo "2. Push initial signals: python -m ta_lab2.scripts.etl.sync_signals_to_vm --full"
echo "3. Push config:          python -m ta_lab2.scripts.etl.sync_config_to_vm"
echo "4. Start executor:       ssh -i $SSH_KEY $VM_USER@$VM_HOST 'sudo systemctl start ta-executor'"
echo "5. Check status:         ssh -i $SSH_KEY $VM_USER@$VM_HOST 'sudo systemctl status ta-executor'"
```

All three scripts should be executable (`chmod +x` in the plan, or note to set the bit).
