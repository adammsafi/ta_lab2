**Header and Selection Controls:**

1. `st.header("Research Explorer")`
2. `st.caption("Explore IC scores, feature predictive power, and market regimes")`

3. Selection sidebar or top bar (use columns for compact layout):
   ```python
   engine = get_engine()
   col1, col2, col3 = st.columns(3)

   with col1:
       assets_df = load_asset_list(engine)
       if assets_df.empty:
           st.warning("No assets found in dim_assets")
           st.stop()
       asset_options = dict(zip(assets_df["symbol"], assets_df["id"]))
       selected_symbol = st.selectbox("Asset", list(asset_options.keys()))
       selected_id = asset_options[selected_symbol]

   with col2:
       tf_list = load_tf_list(engine)
       if not tf_list:
           st.warning("No timeframes found")
           st.stop()
       selected_tf = st.selectbox("Timeframe", tf_list, index=tf_list.index("1D") if "1D" in tf_list else 0)

   with col3:
       feature_names = load_feature_names(engine, selected_id, selected_tf)
       search_query = st.text_input("Search features", "")
       filtered_features = [f for f in feature_names if search_query.lower() in f.lower()] if search_query else feature_names
   ```

4. Feature selector below search:
   ```python
   if filtered_features:
       selected_feature = st.selectbox("Select feature", filtered_features)
   else:
       st.info("No features found. Run IC evaluation first: python -m ta_lab2.scripts.ic.run_ic_eval")
       st.stop()
   ```
