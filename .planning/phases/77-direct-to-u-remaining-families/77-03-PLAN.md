    **Part A: Row count parity verification**

    ```sql
    WITH siloed AS (
        SELECT 'multi_tf' AS alignment_source, COUNT(*) AS n_rows FROM returns_ema_multi_tf
        UNION ALL
        SELECT 'multi_tf_cal_us', COUNT(*) FROM returns_ema_multi_tf_cal_us
        UNION ALL
        SELECT 'multi_tf_cal_iso', COUNT(*) FROM returns_ema_multi_tf_cal_iso
        UNION ALL
        SELECT 'multi_tf_cal_anchor_us', COUNT(*) FROM returns_ema_multi_tf_cal_anchor_us
        UNION ALL
        SELECT 'multi_tf_cal_anchor_iso', COUNT(*) FROM returns_ema_multi_tf_cal_anchor_iso
    ),
    unified AS (
        SELECT alignment_source, COUNT(*) AS n_rows
        FROM returns_ema_multi_tf_u
        GROUP BY alignment_source
    )
    SELECT
        s.alignment_source,
        s.n_rows AS siloed_count,
        COALESCE(u.n_rows, 0) AS unified_count,
        COALESCE(u.n_rows, 0) - s.n_rows AS deficit,
        CASE WHEN COALESCE(u.n_rows, 0) >= s.n_rows THEN 'MATCH' ELSE 'DEFICIT' END AS status
    FROM siloed s
    LEFT JOIN unified u ON s.alignment_source = u.alignment_source
    ORDER BY s.alignment_source;
    ```
