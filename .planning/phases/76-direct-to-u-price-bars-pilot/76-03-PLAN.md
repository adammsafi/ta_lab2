2. Executes this SQL:

```sql
WITH siloed AS (
    SELECT 'multi_tf' AS alignment_source,
           COUNT(*) AS siloed_count
    FROM public.price_bars_multi_tf
    UNION ALL
    SELECT 'multi_tf_cal_us',
           COUNT(*)
    FROM public.price_bars_multi_tf_cal_us
    UNION ALL
    SELECT 'multi_tf_cal_iso',
           COUNT(*)
    FROM public.price_bars_multi_tf_cal_iso
    UNION ALL
    SELECT 'multi_tf_cal_anchor_us',
           COUNT(*)
    FROM public.price_bars_multi_tf_cal_anchor_us
    UNION ALL
    SELECT 'multi_tf_cal_anchor_iso',
           COUNT(*)
    FROM public.price_bars_multi_tf_cal_anchor_iso
),
unified AS (
    SELECT alignment_source,
           COUNT(*) AS u_count
    FROM public.price_bars_multi_tf_u
    GROUP BY alignment_source
)
SELECT
    s.alignment_source,
    s.siloed_count,
    COALESCE(u.u_count, 0) AS u_count,
    CASE
        WHEN s.siloed_count = COALESCE(u.u_count, 0) THEN 'MATCH'
        WHEN COALESCE(u.u_count, 0) > s.siloed_count THEN 'U_HAS_MORE'
        ELSE 'MISMATCH'
    END AS status
FROM siloed s
LEFT JOIN unified u ON u.alignment_source = s.alignment_source
ORDER BY s.alignment_source;
```
