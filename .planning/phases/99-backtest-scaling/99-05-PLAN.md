python -c "

        'SELECT COUNT(DISTINCT strategy_name || asset_id::text || params_json::text || tf || cost_scenario || cv_method) FROM public.strategy_bakeoff_results WHERE tf = %s' % \"'1D'\"

    mc_pop = conn.execute(text(
