def _local_export_csv(engine, table: str, since_ts: datetime | None = None,
                       ts_col: str = "ts") -> str:
    """Export local rows newer than since_ts to CSV string via COPY TO STDOUT."""
    select_sql = f"SELECT * FROM {table}"
    if since_ts is not None:
        select_sql += f" WHERE {ts_col} > '{since_ts.isoformat()}'"
    with engine.raw_connection() as raw:
        cur = raw.cursor()
        buf = io.StringIO()
        cur.copy_expert(f"COPY ({select_sql}) TO STDOUT WITH CSV HEADER", buf)
        return buf.getvalue()

def _push_csv_to_vm(csv_data: str, vm_table: str, timeout: int = 120) -> int:
    """Push CSV data to VM table via SSH psql COPY FROM STDIN.

    if not csv_data.strip():
        return 0

    )
    result = subprocess.run(cmd, input=csv_data, capture_output=True, text=True, timeout=timeout)
    if result.returncode != 0:
        raise RuntimeError(f"SSH COPY push to {vm_table} failed: {result.stderr}")
    # Parse row count from psql output (e.g., "COPY 42")

def sync_table(engine, table: str, ts_col: str = "ts") -> int:
    """Incremental sync of one table local -> VM. Returns rows pushed."""
    vm_wm = _get_vm_watermark(table, ts_col)

**A) Create `run_full_chain.py`**

This is the RECOMMENDED chain approach from RESEARCH.md: a thin wrapper that calls Data -> Features -> Signals -> sync_signals_to_vm in sequence using subprocess. Each pipeline is unaware of the chain -- the wrapper handles sequencing.

```python
"""
Full pipeline chain: Data -> Features -> Signals -> sync_signals_to_vm.

Runs all three local pipelines in sequence, halting on any failure.
This replaces `run_daily_refresh.py --all` with explicit pipeline separation.

Usage:
    python -m ta_lab2.scripts.pipelines.run_full_chain --ids 1,52,825
    python -m ta_lab2.scripts.pipelines.run_full_chain --ids all --dry-run
    python -m ta_lab2.scripts.pipelines.run_full_chain --ids all --no-sync-vms
"""

PIPELINES = [
    ("Data", "ta_lab2.scripts.pipelines.run_data_pipeline"),
    ("Features", "ta_lab2.scripts.pipelines.run_features_pipeline"),
    ("Signals", "ta_lab2.scripts.pipelines.run_signals_pipeline"),
]

def main(argv=None) -> int:
    # Parse args (superset of all pipeline args)
    # --ids, --db-url, --dry-run, --verbose, --continue-on-error
    # --no-sync-vms, --source, -n/--num-processes
    # --skip-stale-check, --staleness-hours
    # --no-sync-signals: skip sync_signals_to_vm at the end

    for name, module in PIPELINES:
        print(f"\n{'=' * 70}")
        print(f"CHAIN: Running {name} pipeline")
        print(f"{'=' * 70}")

        cmd = [sys.executable, "-m", module]
        # Forward relevant args to each pipeline
        # Data gets: --ids, --db-url, --dry-run, --verbose, --continue-on-error,
        #            --no-sync-vms, --source, -n
        # Features gets: --ids, --db-url, --dry-run, --verbose, --continue-on-error,
        #                --skip-stale-check, --staleness-hours, -n
        # Signals gets: --db-url, --dry-run, --verbose, --continue-on-error

        result = subprocess.run(cmd, check=False)
        if result.returncode != 0:
            print(f"\n[CHAIN HALTED] {name} pipeline failed (exit {result.returncode})")
            # Send Telegram alert
            try:
                from ta_lab2.notifications import telegram
                if telegram.is_configured():
                    telegram.send_alert(
                        f"Pipeline Chain HALTED at {name}",
                        f"{name} pipeline failed with exit code {result.returncode}. "
                        f"Subsequent pipelines not started.",
                        severity="warning",
                    )
            except Exception:
                pass
            return result.returncode

    # All 3 pipelines succeeded -- sync signals to VM
    if not args.no_sync_signals and not args.dry_run:
        print(f"\n{'=' * 70}")
        print("CHAIN: Syncing signals to VM")
        print(f"{'=' * 70}")
        sync_cmd = [sys.executable, "-m", "ta_lab2.scripts.etl.sync_signals_to_vm"]
        if args.db_url:
            sync_cmd.extend(["--db-url", args.db_url])
        sync_result = subprocess.run(sync_cmd, check=False)
        if sync_result.returncode != 0:
            print("[WARN] sync_signals_to_vm failed -- signals not pushed to VM")
            # Non-fatal: local pipeline completed, sync is best-effort

    print(f"\n{'=' * 70}")
    print("CHAIN COMPLETE: Data -> Features -> Signals")
    print(f"{'=' * 70}")
    return 0
```
