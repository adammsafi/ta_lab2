- feature_name TEXT NOT NULL (PK)

- promoted_at TIMESTAMPTZ nullable

- promotion_alpha NUMERIC nullable
- promotion_min_pass_rate NUMERIC nullable
- best_ic NUMERIC nullable
- best_horizon INTEGER nullable
- migration_stub_path TEXT nullable
- registered_at TIMESTAMPTZ server_default=now()
- updated_at TIMESTAMPTZ server_default=now()
- Index: idx_feature_registry_lifecycle on lifecycle

**cmc_feature_experiments** (PK: experiment_id UUID gen_random_uuid()):
- experiment_id UUID server_default=gen_random_uuid() (PK)
- feature_name TEXT NOT NULL
- run_at TIMESTAMPTZ server_default=now()
- asset_id INTEGER NOT NULL
- tf TEXT NOT NULL
- train_start TIMESTAMPTZ NOT NULL
- train_end TIMESTAMPTZ NOT NULL
- horizon INTEGER NOT NULL
- horizon_days INTEGER nullable
- return_type TEXT NOT NULL
- regime_col TEXT NOT NULL
- regime_label TEXT NOT NULL
- ic NUMERIC nullable

- ic_p_value NUMERIC nullable

- n_obs INTEGER nullable

- yaml_digest TEXT nullable

- Index: idx_feature_experiments_feature_name on feature_name
- Index: idx_feature_experiments_run_at on run_at
