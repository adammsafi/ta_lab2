**Change 1: Add `_load_l4_regime_for_asset()` helper method to PaperExecutor**

Add this method in the internal helpers section (after `_load_active_configs`, around line 220):

```python
    def _load_l4_regime_for_asset(
        self, conn, asset_id: int
    ) -> tuple[Optional[str], float, float]:
        """
        Read the latest L4 macro regime label and gross_cap from cmc_regimes
        for the given asset.

        Returns:
            (l4_label, gross_cap, size_mult) tuple.
            If no regime row exists or l4_label is NULL, returns (None, 1.0, 1.0).
        """
        try:
            row = conn.execute(
                text(
                    """
                    SELECT l4_label, gross_cap, size_mult
                    FROM public.cmc_regimes
                    WHERE id = :asset_id AND tf = '1D'
                    ORDER BY ts DESC
                    LIMIT 1
                    """
                ),
                {"asset_id": asset_id},
            ).fetchone()
        except Exception as exc:  # noqa: BLE001
            self.logger.debug(
                "_load_l4_regime: query failed for asset_id=%d: %s",
                asset_id,
                exc,
            )
            return (None, 1.0, 1.0)

        if row is None:
            return (None, 1.0, 1.0)

        l4_label = row.l4_label  # may be None if L4 is disabled
        gross_cap = float(row.gross_cap) if row.gross_cap is not None else 1.0
        size_mult = float(row.size_mult) if row.size_mult is not None else 1.0

        return (l4_label, gross_cap, size_mult)
```
