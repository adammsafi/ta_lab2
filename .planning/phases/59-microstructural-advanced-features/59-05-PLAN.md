```python
def refresh_microstructure(
    engine, ids, start, end, tf="1D", alignment_source="multi_tf",
) -> RefreshResult:
    """Refresh microstructure columns in cmc_features for given tf."""
    from ta_lab2.scripts.features.microstructure_feature import (
        MicrostructureFeature, MicrostructureConfig,
    )
    table = "cmc_features (microstructure)"
    t0 = time.time()
    try:
        config = MicrostructureConfig(tf=tf, alignment_source=alignment_source)
        feature = MicrostructureFeature(engine, config)
        rows_written = feature.compute_for_ids(ids=ids, start=start, end=end)
        return RefreshResult(table=table, rows_inserted=rows_written,
                           duration_seconds=time.time() - t0, success=True)
    except Exception as e:
        logger.error(f"Microstructure refresh failed (tf={tf}): {e}", exc_info=True)
        return RefreshResult(table=table, rows_inserted=0,
                           duration_seconds=time.time() - t0, success=False, error=str(e))
```

```python
def _load_sadf_flags(engine: Engine, asset_id: int) -> pd.Series:
    """Load sadf_is_explosive flags from cmc_features for the given asset.

    Returns a Series indexed by ts with boolean values.
    Returns empty Series if no SADF data available (graceful fallback).
    """
    query = text("""
        SELECT ts, sadf_is_explosive
        FROM cmc_features
        WHERE id = :id AND tf = '1D' AND sadf_is_explosive IS NOT NULL
        ORDER BY ts
    """)
    try:
        df = pd.read_sql(query, engine, params={"id": asset_id})
        if df.empty:
            return pd.Series(dtype=bool)
        df["ts"] = pd.to_datetime(df["ts"], utc=True)
        return df.set_index("ts")["sadf_is_explosive"]
    except Exception as exc:
        logger.warning("  SADF load failed for id=%s: %s", asset_id, exc)
        return pd.Series(dtype=bool)
```

```python
from ta_lab2.config import TARGET_DB_URL
from sqlalchemy import create_engine, text
import pandas as pd

e = create_engine(TARGET_DB_URL)
with e.connect() as c:
    # Get latest codependence snapshot
    codep = pd.read_sql(text(\"\"\"
        SELECT id_a, id_b, pearson_corr, distance_corr, mutual_info, variation_of_info, n_obs
        FROM cmc_codependence
        WHERE tf = '1D'
        ORDER BY computed_at DESC
        LIMIT 50
    \"\"\"), c)

    if codep.empty:
        print('No codependence data found. Run codependence_feature first.')
    else:
        # Show non-linear vs Pearson divergence
        codep['dcor_pearson_gap'] = (codep['distance_corr'] - codep['pearson_corr'].abs()).round(4)
        print('=== Codependence vs Pearson Comparison ===')
        print(f'Pairs analyzed: {len(codep)}')
        print()
        print('Top pairs where distance_corr > |pearson_corr| (non-linear dependence detected):')
        divergent = codep.nlargest(5, 'dcor_pearson_gap')
        for _, row in divergent.iterrows():
            print(f'  ({int(row.id_a)}, {int(row.id_b)}): pearson={row.pearson_corr:.4f}  dcor={row.distance_corr:.4f}  MI={row.mutual_info:.4f}  gap={row.dcor_pearson_gap:+.4f}')
        print()
        print('Summary statistics:')
        print(f'  Mean |Pearson|:         {codep.pearson_corr.abs().mean():.4f}')
        print(f'  Mean distance_corr:     {codep.distance_corr.mean():.4f}')
```
