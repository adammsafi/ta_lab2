**Main flow:**
```python
def main() -> int:
    args = parse_args()
    setup_logging(args.verbose)

    logger.info("Starting macro regime classification...")

    # Load config
    config = load_macro_regime_config(args.config)
    profile = args.profile or config["active_profile"]
    logger.info(f"Using profile: {profile}")

    # Get engine
    engine = get_engine()

    # Create classifier
    classifier = MacroRegimeClassifier(config, engine)

    # Classify
    df = classifier.classify(
        start_date=args.start_date,
        end_date=args.end_date,
        profile=profile,
        full=args.full,
    )

    if df.empty:
        logger.info("No new regime rows to write (up to date)")
        return 0

    logger.info(f"Computed {len(df)} regime rows ({df.index.min()} to {df.index.max()})")

    # Summary stats
    if "macro_state" in df.columns:
        state_counts = df["macro_state"].value_counts()
        logger.info(f"Macro state distribution:\n{state_counts.to_string()}")

    if args.dry_run:
        print(f"[DRY RUN] Would upsert {len(df)} rows to cmc_macro_regimes (profile={profile})")
        print(df.tail(10).to_string())
        return 0

    # Upsert
    count = classifier.upsert_results(df)
    logger.info(f"Upserted {count} rows to cmc_macro_regimes")

    return 0
```
