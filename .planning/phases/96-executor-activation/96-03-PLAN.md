    **2. Add `--seed-watermarks` flag to `seed_executor_config.py`**:

    Add a new CLI argument `--seed-watermarks` that, after seeding configs, sets `last_processed_signal_ts = MAX(ts)` for each active config where watermark is NULL:

    ```python
    def seed_watermarks(db_url: str) -> dict:
        """
        Set watermarks to MAX(ts) per signal table for active configs with NULL watermark.
        Prevents historical signal replay on first executor run.
        """
        engine = create_engine(db_url, poolclass=NullPool)
        summary = {"watermarks_set": 0, "already_set": 0, "no_signals": 0}

        try:
            with engine.begin() as conn:
                # Load active configs with NULL watermark
                rows = conn.execute(text(
                    "SELECT config_id, signal_type, signal_id, last_processed_signal_ts "
                    "FROM dim_executor_config WHERE is_active = TRUE"
                )).fetchall()

                for row in rows:
                    if row.last_processed_signal_ts is not None:
                        summary["already_set"] += 1
                        continue

                    # Import SIGNAL_TABLE_MAP to resolve table name
                    from ta_lab2.executor.signal_reader import SIGNAL_TABLE_MAP
                    signal_table = SIGNAL_TABLE_MAP.get(row.signal_type)
                    if not signal_table:
                        print(f"[SKIP] Unknown signal_type '{row.signal_type}' for config_id={row.config_id}")
                        continue

                    # Get MAX(ts) from signal table
                    max_ts_row = conn.execute(text(
                        f"SELECT MAX(ts) AS max_ts FROM public.{signal_table} "
                        "WHERE signal_id = :signal_id"
                    ), {"signal_id": row.signal_id}).fetchone()

                    if max_ts_row and max_ts_row.max_ts is not None:
                        conn.execute(text(
                            "UPDATE dim_executor_config SET last_processed_signal_ts = :max_ts "
                            "WHERE config_id = :config_id"
                        ), {"max_ts": max_ts_row.max_ts, "config_id": row.config_id})
                        print(f"[OK] Watermark set for config_id={row.config_id}: {max_ts_row.max_ts}")
                        summary["watermarks_set"] += 1
                    else:
                        print(f"[INFO] No signals yet for config_id={row.config_id} ({row.signal_type})")
                        summary["no_signals"] += 1
        finally:
            engine.dispose()

        return summary
    ```

    Wire this into `main()`:
    - Add `--seed-watermarks` as a new CLI flag
    - After seeding configs (if `--seed-watermarks` is set), call `seed_watermarks(db_url)`
    - Print watermark summary after config summary

    The `f"public.{signal_table}"` usage is safe because signal_table comes from `SIGNAL_TABLE_MAP` which is a hardcoded dict (no SQL injection risk -- same pattern as signal_reader.py).

    In `PositionSizer.compute_target_position()`, add a new branch for `sizing_mode == "bl_weight"`:

    ```python
    elif sizing_mode == "bl_weight":
        # Look up most recent BL weight from portfolio_allocations.
        # Caller MUST pass conn and asset_id via kwargs.
        conn = kwargs.get("conn")
        asset_id = kwargs.get("asset_id")
        if conn is not None and asset_id is not None:
            bl_sql = text(
                "SELECT COALESCE(final_weight, weight) AS bl_weight "
                "FROM public.portfolio_allocations "
                "WHERE asset_id = :asset_id AND optimizer = 'bl' "
                "ORDER BY ts DESC LIMIT 1"
            )
            row = conn.execute(bl_sql, {"asset_id": asset_id}).fetchone()
            if row and row.bl_weight is not None and float(row.bl_weight) > 0:
                fraction = Decimal(str(row.bl_weight))
                logger.debug(
                    "compute_target_position: bl_weight=%.6f for asset_id=%s",
                    float(fraction), asset_id,
                )
            else:
                # BL de-selected or no BL row: close position (return 0)
                logger.info(
                    "compute_target_position: bl_weight=0 or missing for asset_id=%s, returning flat",
                    asset_id,
                )
                return Decimal("0")
        else:
            # No connection: fall back to fixed_fraction
            logger.warning(
                "compute_target_position: bl_weight mode but no conn/asset_id, "
                "falling back to fixed_fraction"

            # Use default fixed_fraction
    ```
