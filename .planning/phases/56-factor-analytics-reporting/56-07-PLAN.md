    Step 6: Compute and save MAE/MFE
    ```python
    # 6. Compute MAE/MFE for closed trades
    try:
        from ta_lab2.analysis.mae_mfe import compute_mae_mfe, _load_close_prices
        if not result.trades_df.empty:
            close_prices = _load_close_prices(self.engine, result.asset_id, result.start_ts, result.end_ts)
            trades_with_excursion = compute_mae_mfe(result.trades_df, close_prices)
            # UPDATE trades with mae/mfe values
            for _, trade in trades_with_excursion.iterrows():
                if trade.get('mae') is not None:
                    conn.execute(text(
                        "UPDATE public.cmc_backtest_trades SET mae = :mae, mfe = :mfe "
                        "WHERE run_id = :run_id AND entry_ts = :entry_ts"
                    ), {"mae": float(trade['mae']), "mfe": float(trade['mfe']),
                        "run_id": result.run_id, "entry_ts": trade['entry_ts']})
            logger.info(f"MAE/MFE computed for {len(trades_with_excursion)} trades")
    except Exception as e:
        logger.warning(f"MAE/MFE computation failed (non-fatal): {e}")
    ```

    Step 7: Monte Carlo CI
    ```python
    # 7. Monte Carlo trade resampling
    try:
        from ta_lab2.analysis.monte_carlo import monte_carlo_trades
        if not result.trades_df.empty:
            mc_result = monte_carlo_trades(result.trades_df)
            if mc_result['mc_sharpe_lo'] is not None:
                conn.execute(text(
                    "UPDATE public.cmc_backtest_metrics SET "
                    "mc_sharpe_lo = :lo, mc_sharpe_hi = :hi, mc_sharpe_median = :med, mc_n_samples = :n "
                    "WHERE run_id = :run_id"
                ), {"lo": mc_result['mc_sharpe_lo'], "hi": mc_result['mc_sharpe_hi'],
                    "med": mc_result['mc_sharpe_median'], "n": mc_result['mc_n_samples'],
                    "run_id": result.run_id})
                logger.info(f"MC Sharpe CI: [{mc_result['mc_sharpe_lo']:.2f}, {mc_result['mc_sharpe_hi']:.2f}]")
    except Exception as e:
        logger.warning(f"Monte Carlo computation failed (non-fatal): {e}")
    ```
