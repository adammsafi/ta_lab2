```python
if builder.name.startswith("1d_"):
    # All 1D builders use the generalized script with --source flag
    cmd.extend(["--db-url", db_url])
    cmd.extend(["--ids", ids])
    if full_rebuild:
        cmd.append("--full-rebuild")
    # Pass --source from custom_args
    source = (builder.custom_args or {}).get("source")
    if source:
        cmd.extend(["--source", source])
    # CMC gets --keep-rejects for visibility
    if source == "cmc":
        cmd.append("--keep-rejects")
```
